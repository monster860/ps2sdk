244int _start( int argc, char *argv[] )
245{
246 M_PRINTF( "Copyright (c) 2004 adresd\n" );
247
248 if ( argc != 2 )
249 {
250 M_PRINTF( "HOST requires based argument\n" );
251 return MODULE_NO_RESIDENT_END;
252 }
253 else
254 {
255 // Copy the base location.
256 strncpy( base, argv[1] ,sizeof(base) - 1);
257 base[sizeof(base) - 1] = '\0';
258
259 M_PRINTF( "redirecting '%s:' to '%s'\n",FS_REPNAME,base);
260 fd_global = 1;
261
262 M_PRINTF( "HOST final step, bye\n" );
263
264 // Install naplink RPC handler
265 naplinkRpcInit();
266
267 // now install the fileio driver
268 io_DelDrv( FS_REPNAME );
269 io_AddDrv( &driver );
270 }
271 return MODULE_RESIDENT_END;
