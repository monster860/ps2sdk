23int _start(int argc, char *argv[])
24{
25 int *BootMode;
26 int thread_id;
27 iop_thread_t thread_param;
28
29 FlushDcache();
30 BootMode = QueryBootMode(3);
31 if (BootMode) {
32 int iop_boot_param;
33
34 iop_boot_param = BootMode[1];
35 if ((iop_boot_param & 1) != 0) {
36 printf(" No SIF service(loadfile)\n");
37 return MODULE_NO_RESIDENT_END;
38 }
39 if ((iop_boot_param & 2) != 0) {
40 printf(" No LoadFile service\n");
41 return MODULE_NO_RESIDENT_END;
42 }
43 }
44 CpuEnableIntr();
45 thread_param.attr = 0x2000000;
46 thread_param.thread = loadfile_rpc_start_thread;
47 thread_param.priority = 88;
48 thread_param.stacksize = 4096;
49 thread_param.option = 0;
50 thread_id = CreateThread(&thread_param);
51 if (thread_id <= 0) {
52 return MODULE_NO_RESIDENT_END;
53 }
54 StartThread(thread_id, 0);
55 return MODULE_RESIDENT_END;
56}
