57__attribute__((weak))
58void StartTimerSystemTime(void)
59{
60 u32 oldintr;
61
62 oldintr = DIntr();
63 __time_intr_overflow_count = 0;
64
65 ee_kmode_enter();
66 *T_MODE_WR = 0x0000; // Disable T_MODE
67 // Initialize the timer registers
68 // CLKS: 0x02 - 1/256 of the BUSCLK (0x01 is 1/16th)
69 // CUE: 0x01 - Start/Restart the counting
70 // OVFE: 0x01 - An interrupt is generated when an overflow occurs
71 *T_COUNT_WR = 0;
72 *T_MODE_WR = Tn_MODE(0x02, 0, 0, 0, 0, 0x01, 0, 0x01, 0, 0);
73 ee_kmode_exit();
74
75 if (__time_intr_overflow_id == -1)
76 {
77 __time_intr_overflow_id = AddIntcHandler(INTC_TIM, intrOverflow, 0);
78 EnableIntc(INTC_TIM);
79 }
80
81 if(oldintr) { EIntr(); }
82}

114#ifdef F_iGetTimerSystemTime
123u64 iGetTimerSystemTime(void)
124{
125 u64 t;
126
127 // Tn_COUNT is 16 bit precision. Therefore, each __time_intr_overflow_count is 65536 ticks
128 t = *T_COUNT_RD + (__time_intr_overflow_count << 16);
129
130 // check if the overflow was produced but the interruption callback isn't be executed yet
131 if (((*T_MODE_RD) & (1 << 11)) != 0) {
132 t += (1 << 16);
133 }
134
135 return (t << 8); // Because counter clock config was 0x02 (which means BUSCLK/256)
136}
