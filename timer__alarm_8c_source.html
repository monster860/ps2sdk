39#ifdef F___Timer2Resched
40void __Timer2Resched(struct timer_alarm_t *alarm) {
41 if (!alarm) {
42 // Disable the counter, no alarms are queued
43 *T_MODE = 0x0000;
44 } else {
45 // Setup timer for the next event.
46 s64 wait_time = (s64)alarm->scheduled_time - (s64)(iGetTimerSystemTime());
47 u64 alarm_clocks = wait_time <= 0 ? 1 : wait_time;
48
49 if (alarm_clocks < 65536) { // [0..2^16)
50 // Using /1 clock mode for better precision
51 *T_COUNT = 0;
52 *T_COMP = (u16)alarm_clocks;
53 *T_MODE = Tn_MODE(0, 0, 0, 0, 0, 1, 1, 0, 1, 1);
54 }
55 else if (alarm_clocks < 1048576) { // [2^16..2^20)
56 // Using /16 mode
57 *T_COUNT = 0;
58 *T_COMP = (u16)(alarm_clocks >> 4);
59 *T_MODE = Tn_MODE(1, 0, 0, 0, 0, 1, 1, 0, 1, 1);
60 }
61 else if (alarm_clocks < 16777216) { // [2^20..2^24)
62 // Using /256 mode
63 *T_COUNT = 0;
64 *T_COMP = (u16)(alarm_clocks >> 8);
65 *T_MODE = Tn_MODE(2, 0, 0, 0, 0, 1, 1, 0, 1, 1);
66 }
67 else {
68 // Requires several interrups, there's no slower clock mode
69 // Using counter overflow mode
70 *T_COUNT = 0;
71 *T_MODE = Tn_MODE(2, 0, 0, 0, 0, 1, 0, 1, 1, 1);
72 }
73 }
74}
75#else
76extern void __Timer2Resched(struct timer_alarm_t *alarm);
77#endif

79#ifdef F_InitTimerAlarm
80static int timOverflow(int ca) {
81 (void)ca;
82
83 // Check if we actually triggered an alarm
84 if (__event_queue) {
85 if (GetTimerSystemTime() > (__event_queue->scheduled_time)) {
86 // Remove the top of the queue
87 struct timer_alarm_t *gone = __event_queue;
88 __event_queue = __event_queue->next;
89 if (__event_queue)
90 __event_queue->prev = NULL;
91 gone->prev = NULL;
92 gone->next = NULL;
93 gone->scheduled_time = 0;
94 __Timer2Resched(__event_queue);
95
96 if (gone->callback)
97 gone->callback(gone, gone->usr_arg);
98 }
99 else // Timer might require several IRQs (for timers over ~100ms)
100 __Timer2Resched(__event_queue);
101 }
102
103 // Clear both compare and overflow flags (since we can use either)
104 *T_MODE |= (3 << 10);
105
106 ExitHandler();
107 return -1;
108}

149#ifdef F_iStopTimerAlarm
153void iStopTimerAlarm(struct timer_alarm_t *alarm) {
154 // Bail if the alarm is not set.
155 if (!alarm->scheduled_time)
156 return;
157
158 // Check if the event is the next event in the queue.
159 if (__event_queue == alarm) {
160 // Remove it from the front of the queue
161 if (alarm->next)
162 alarm->next->prev = NULL;
163 __event_queue = alarm->next;
164
165 // Reschedule next event
166 __Timer2Resched(__event_queue);
167 }
168 else {
169 // Simply remove it, since it is not yet scheduled.
170 alarm->prev->next = alarm->next;
171 if (alarm->next)
172 alarm->next->prev = alarm->prev;
173 }
174
175 // Cleanup the structure
176 alarm->next = NULL;
177 alarm->prev = NULL;
178 alarm->scheduled_time = 0;
179}
180#endif

199#ifdef F_iStartTimerAlarm
203void iStartTimerAlarm(struct timer_alarm_t *alarm) {
204 // Bail if the alarm is already set!
205 if (alarm->scheduled_time)
206 return;
207
208 // Calculate clock scheduling time
209 u64 sched_time = iGetTimerSystemTime() + alarm->timer_cycles;
210 alarm->scheduled_time = sched_time;
211
212 if (!__event_queue) {
213 // No alarm was set
214 __event_queue = alarm;
215 alarm->next = alarm->prev = NULL;
216 __Timer2Resched(alarm);
217 }
218 else if (__event_queue->scheduled_time > alarm->scheduled_time) {
219 // This alarm is the next alarm to trigger
220 alarm->next = __event_queue;
221 alarm->prev = NULL;
222 __event_queue->prev = alarm;
223 __event_queue = alarm;
224 __Timer2Resched(alarm);
225 }
226 else {
227 // Just queue it at the right place (keep the list sorted)
228 struct timer_alarm_t *it = __event_queue;
229 for (; it != NULL; it = it->next) {
230 if (!it->next || it->next->scheduled_time > alarm->scheduled_time) {
231 alarm->next = it->next;
232 alarm->prev = it;
233 if (it->next)
234 it->next->prev = alarm;
235 it->next = alarm;
236 break;
237 }
238 }
239 // No need to update the alarm at all
240 }
241}
242#endif

288void ThreadWaitClock(u64 clock_cycles) {
289 // Setup a semaphore and alarm. The alarm will unblock the semaphore.
290 ee_sema_t sema = {.max_count = 1, .init_count = 0, .option = 0};
291 s32 sema_id = CreateSema(&sema);
292 struct timer_alarm_t alarm;
293 InitializeTimerAlarm(&alarm);
294 SetTimerAlarm(&alarm, clock_cycles, _wake_sema_cb, &sema_id);
295 StartTimerAlarm(&alarm);
296
297 WaitSema(sema_id);
298
299 DeleteSema(sema_id);
300}
